{
		int a;

		long long int c;
		char d;
		float f;

		printf("Size of a char: %lu byte(s)\n", (unsigned long)sizeof(d));

		printf("Size of a long int: %lu byte(s)\n", (unsigned long)sizeof(b));

		printf("Size of a float: %lu byte(s)\n", (unsigned long)sizeof(f));
		return (0);
}
